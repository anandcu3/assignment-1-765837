# Install docker and run official mongodb image

docker run -d -p 27017:27017 --name m1 mongo

# Here, one of the important part is -p option: -p 27017:27107 exposes port 27017 so we can connect to mongodb instance from our local machine.

# Then create a virtual environment on your local machine. For this I'm using anaconda to help me manage the various environments

conda create --name big_data python=3.5
activate big_data
pip install pymongo
